16  package org.mybatis.dynamic.sql.where.condition;
17  
18  import static org.mybatis.dynamic.sql.util.StringUtilities.spaceAfter;
19  
20  import java.util.Arrays;
21  import java.util.Collection;
22  import java.util.function.UnaryOperator;
23  import java.util.stream.Collectors;
24  import java.util.stream.Stream;
25  
26  import org.mybatis.dynamic.sql.AbstractListValueCondition;
27  import org.mybatis.dynamic.sql.Callback;
28  
29  public class IsNotIn<T> extends AbstractListValueCondition<T, IsNotIn<T>> {
30  
31      protected IsNotIn(Builder<T> builder) {
32          super(builder);
33      }
34  
35      @Override
36      public String renderCondition(String columnName, Stream<String> placeholders) {
37          return spaceAfter(columnName)
38                  + placeholders.collect(
39                          Collectors.joining(",", "not in (", ")")); //$NON-NLS-1$ //$NON-NLS-2$ //$NON-NLS-3$
40      }
41  
42      @Override
43      public IsNotIn<T> withListEmptyCallback(Callback callback) {
44          return new Builder<T>()
45                  .withValues(values)
46                  .withValueStreamTransformer(valueStreamTransformer)
47                  .withEmptyCallback(callback)
48                  .build();
49      }
50  
51      /**
52       * This method allows you to modify the condition's values before they are placed into the parameter map.
53       * For example, you could filter nulls, or trim strings, etc. This process will run before final rendering of SQL.
54       * If you filter values out of the stream, then final condition will not reference those values. If you filter all
55       * values out of the stream, then the condition will not render.
56       *
57       * @param valueStreamTransformer a UnaryOperator that will transform the value stream before
58       *     the values are placed in the parameter map
59       * @return new condition with the specified transformer
60       */
61      public IsNotIn<T> then(UnaryOperator<Stream<T>> valueStreamTransformer) {
62          return new Builder<T>()
63                  .withValues(values)
64                  .withValueStreamTransformer(valueStreamTransformer)
65                  .withEmptyCallback(emptyCallback)
66                  .build();
67      }
68  
69      @SafeVarargs
70      public static <T> IsNotIn<T> of(T... values) {
71          return of(Arrays.asList(values));
72      }
73  
74      public static <T> IsNotIn<T> of(Collection<T> values) {
75          return new Builder<T>()
76                  .withValues(values)
77                  .build();
78      }
79  
80      public static class Builder<T> extends AbstractListConditionBuilder<T, Builder<T>> {
81          @Override
82          protected Builder<T> getThis() {
83              return this;
84          }
85  
86          public IsNotIn<T> build() {
87              return new IsNotIn<>(this);
88          }
89      }
90  }
